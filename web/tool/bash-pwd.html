    # bash - 获取脚本的运行目录

    如何在 bash 脚本里面获取到当前的运行目录，免得写死路径，导致不同环境使用的时候，都要修改脚本。 判断当前文件是否是软链接文件，若是则获取真正的文件；

    #!/bin/bash
if [ -L $0 ]
then
    BASE_DIR=`dirname $(readlink $0)`
else
    BASE_DIR=`dirname $0`
fi
basepath=$(cd $BASE_DIR; pwd)
echo $basepath
