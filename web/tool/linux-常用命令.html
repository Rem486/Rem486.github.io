# 1.查看进程和终止进程

  1. ps 查看进程，找到进程 pid
ps -ef | grep node
# 查看所有进程里 CMD 是 node 的进程信息

ps -aux | grep node
# -aux 显示所有包含其他使用者的进程
# ps 命令用于查看当前正在运行的进程

# 2.查看日志

# 查询最后100行日志
tail -n 100 info.log

# 查询从100行之后的所有日志
tail -n +100 info.log

# 查询日志文件中的头10行日志;
head -n 10 info.log

# 查询日志文件除了最后10行的其他所有日志;
head -n -10 info.log

# 获取关键字的行号
# -n 显示行号
cat -n test.log |grep "关键字"

# 查询超多日志
# 1. 使用 more 和 less 命令，分页打印，然后空格翻页
cat -n info.log |grep "_id" |more
# 2. 使用 > xxx.txt 把输出的内容导入文件，然后拉下文件分析
cat -n info.log |grep "_id" >xxx.txt
